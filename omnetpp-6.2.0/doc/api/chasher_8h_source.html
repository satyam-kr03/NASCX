45  void merge(uint32_t x) {
46  // rotate value left by one bit, and xor with new data
47  value = ((value << 1) | (value >> 31)) ^ x;
48  }

50  void merge2(uint64_t x) {
51  merge((uint32_t)x);
52  merge((uint32_t)(x>>32));
53  }
